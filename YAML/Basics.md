```bash
    name: john
    gender: male
    age: 24
    address: # dictionary
        city: edison
        state: 'new jersey'
        country: 'united states'
    payslips: # lists
        - # 1
            month: june
            amount: 1400
        - # 2
            month: july
            amount: 2400
        - # 3
            month: august
            amount: 3400

```
